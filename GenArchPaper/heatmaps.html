h2TS<-read.table(my.dir %&% "GTEx_Tissue-Specific_local_h2.txt",header=T)

h2TS <- arrange(h2TS, CrossT)
matTS <- as.matrix(h2TS[,2:dim(h2TS)[2]])
heatmap(matTS,Rowv=NA,col = cm.colors(256),margins = c(5, 0),labRow=NA)
